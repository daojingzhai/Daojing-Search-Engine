<p>子查询与连接在多表操作中的使用</p>
<p>插入<br /></p>
<pre><code class="language-sql"> INSERT tdb_goods_cates(cate_name) SELECT goods_cate FROM tdb_goods GROUP BY goods_cate;</code></pre><br /><strong> 多表更新</strong><br />
 UPDATE table_reference<br />
 SET col_name1={expr|DEFAULT}<br />
 [,col_name1={expr|DEFAULT}]...<br />
 [WHERE where_condition]<br /><p> *table_reference:表的参照关系</p>
 内连接，外连接，左连接，右连接<br /><pre><code class="language-sql">UPDATE tdb_goods INNER JOIN tdb_goods_cates ON goods_cate=cate_name SET goods_cate=cate_id;</code></pre> 有相同列名时<br /><pre><code class="language-sql">  UPDATE tdb_goods INNER JOIN tdb_goods_brands ON
    -&gt; tdb_goods.brand_name=tdb_goods_brands.brand_name
    -&gt; SET tdb_goods.brand_name=tdb_goods_brands.brand_id;
   UPDATE tdb_goods AS  g INNER JOIN tdb_goods_brands AS b ON
    -&gt; g.brand_name=b.brand_name
    -&gt; SET g.brand_name=b.brand_id;</code></pre> <br />
 创建表<br />
 CREATE TABLE [IF NOT EXISTS] table_name(<br />
 )select_statement<br /><pre><code class="language-sql">  CREATE TABLE tdb_goods_brands(
    -&gt; brand_id SMALLINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
    -&gt; brand_name VARCHAR(40) NOT NULL
    -&gt; )
    -&gt; SELECT brand_name FROM tdb_goods GROUP BY brand_name;

      CREATE TABLE tdb_goods_brands(
    -&gt; brand_id SMALLINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
    -&gt; brand_name VARCHAR(40) NOT NULL
    -&gt; )
    -&gt; SELECT brand_name FROM tdb_goods GROUP BY brand_name;</code></pre><br />
做一下修改~~    <br />
================================================================<br /><pre><code class="language-html">    ALTER TABLE tdb_goods
    -&gt; CHANGE goods_cate cate_id SMALLINT UNSIGNED NOT NULL,
    -&gt; CHANGE brand_name brand_id SMALLINT UNSIGNED NOT NULL;    </code></pre>=================================================================<br /><br /><br /><strong>自身连接（用于无限分类数据表）</strong><br />
-- 无限分类的数据表设计<br /><pre><code class="language-sql">   CREATE TABLE tdb_goods_types(
     type_id   SMALLINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
     type_name VARCHAR(20) NOT NULL,
     parent_id SMALLINT UNSIGNED NOT NULL DEFAULT 0 --父级类型
  );
 SELECT s.type_id,s.type_name,p.type_name FROM tdb_goods_types AS s JOIN tdb_goods_types AS p
    -&gt; ON s.parent_id = p.type_id;    </code></pre>    <br /><strong>多表删除重复记录</strong><br /><pre><code class="language-html">DELETE t1 FROM tdb_goods AS t1 LEFT JOIN (
    -&gt; SELECT goods_name,goods_id FROM tdb_goods GROUP BY goods_name HAVING COUNT(goods_name)&gt;=2) AS t2
    -&gt; ON t1.goods_name=t2.goods_name
    -&gt; WHERE t1.goods_id&gt;t2.goods_id;</code></pre><br /><br />
